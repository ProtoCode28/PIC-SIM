public void BCF(int cmd)
        {
            int result = GetData(cmd);
            int a = ExtractBitB(cmd);   // zb 7 also bit an 7. stelle muss auf 0 gesetzt werden
            int b = 0;
            if (a == 7)
            {
                b = 0b0111_1111;
            }
            else if (a == 6)
            {
                b = 0b1011_1111;
            }
            else if (a == 5)
            {
                b = 0b1101_1111;
            }
            else if (a == 4)
            {
                b = 0b1110_1111;
            }
            else if (a == 3)
            {
                b = 0b1111_0111;
            }
            else if (a == 2)
            {
                b = 0b1111_1011;
            }
            else if (a == 1)
            {
                b = 0b1111_1101;
            }
            else if (a == 0)
            {
                b = 0b1111_1110;
            }
            result &= b;
            SetData(cmd, result);
        }
